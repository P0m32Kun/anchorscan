# AnchorScan v1.4 Audit Export Filter Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add fixed severity multi-select filters, default scan audit artifacts, and richer downloadable report exports for v1.4.

**Architecture:** Keep the current scan/report/store flow. Add `artifact_dir` to scan runs, write tool outputs into one run directory during `RunScan`, and expose report downloads through the existing `/reports/` handler. Reuse existing report evidence fields instead of adding a larger artifact system.

**Tech Stack:** Go standard library, `modernc.org/sqlite`, existing `net/http` templates, existing app/store/report packages.

## Global Constraints

- No PDF export in v1.4.
- No new external dependencies.
- Default artifact root is `data/artifacts` for CLI scans and `<managed data root>/artifacts` for web scans.
- Each scan creates one app-owned run directory: `<artifact_root>/<run_id>/`.
- Delete only the run artifact directory, never the user-selected artifact root.
- Supported severities are exactly: `critical`, `high`, `medium`, `low`, `info`.
- Unknown report export format returns HTTP 400.
- Run `go test ./...` before final completion.

```go
func (s *Store) ListProjectArtifactDirs(projectID string) ([]string, error) {
	rows, err := s.db.Query(
		`SELECT artifact_dir
		 FROM scan_runs
		 WHERE project_id = ? AND artifact_dir != ''
		 ORDER BY started_at ASC, run_id ASC`,
		projectID,
	)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var dirs []string
	for rows.Next() {
		var dir string
		if err := rows.Scan(&dir); err != nil {
			return nil, err
		}
		dirs = append(dirs, dir)
	}
	return dirs, rows.Err()
}
```

```go
func DiscoverPortsWithOutput(ctx context.Context, runner Runner, binaryPath string, target string, ports string, extraArgs []string) ([]int, []byte, error) {
	args := []string{"-a", target}
	if strings.Contains(ports, "-") && !strings.Contains(ports, ",") {
		args = append(args, "--range", ports)
	} else {
		args = append(args, "--ports", ports)
	}
	args = append(args, "-g", "--no-banner")
	args = append(args, extraArgs...)

	out, err := runner.Run(ctx, binaryPath, args)
	if err != nil {
		return nil, out, withOutputError(err, out)
	}

	matches := extractPortMatches(string(out))
	seen := map[int]struct{}{}
	var found []int
	for _, match := range matches {
		port, err := strconv.Atoi(match)
		if err != nil {
			continue
		}
		if _, ok := seen[port]; ok {
			continue
		}
		seen[port] = struct{}{}
		found = append(found, port)
	}

	sort.Ints(found)
	return found, out, nil
}
```

```go
package app

import (
	"crypto/sha1"
	"encoding/hex"
	"os"
	"path/filepath"
	"strings"
	"unicode"
)

func safeArtifactName(parts ...string) string {
	joined := strings.Join(parts, "-")
	joined = strings.TrimSpace(joined)
	var b strings.Builder
	lastDash := false
	for _, r := range joined {
		ok := unicode.IsLetter(r) || unicode.IsDigit(r) || r == '.' || r == '_' || r == ',' || r == '-'
		if ok {
			b.WriteRune(r)
			lastDash = false
			continue
		}
		if !lastDash {
			b.WriteByte('-')
			lastDash = true
		}
	}
	name := strings.Trim(b.String(), "-.")
	if name == "" {
		return "artifact"
	}
	return name
}

func writeArtifact(dir, name string, data []byte) (string, error) {
	if dir == "" || len(data) == 0 {
		return "", nil
	}
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return "", err
	}
	base := safeArtifactName(name)
	path := filepath.Join(dir, base)
	if _, err := os.Stat(path); os.IsNotExist(err) {
		return path, os.WriteFile(path, data, 0o644)
	}
	sum := sha1.Sum([]byte(base + string(data)))
	ext := filepath.Ext(base)
	stem := strings.TrimSuffix(base, ext)
	path = filepath.Join(dir, stem+"-"+hex.EncodeToString(sum[:])[:8]+ext)
	return path, os.WriteFile(path, data, 0o644)
}
```

```go
var severityOptions = []string{"critical", "high", "medium", "low", "info"}

func parseSeverities(values url.Values) []string {
	allowed := map[string]struct{}{}
	for _, option := range severityOptions {
		allowed[option] = struct{}{}
	}
	seen := map[string]struct{}{}
	var out []string
	for _, raw := range values["severity"] {
		for _, part := range strings.Split(raw, ",") {
			value := strings.ToLower(strings.TrimSpace(part))
			if _, ok := allowed[value]; !ok {
				continue
			}
			if _, ok := seen[value]; ok {
				continue
			}
			seen[value] = struct{}{}
			out = append(out, value)
		}
	}
	return out
}

func severitySelected(selected []string, value string) bool {
	for _, item := range selected {
		if item == value {
			return true
		}
	}
	return false
}

func findingMatchesSeverity(finding report.Finding, selected []string) bool {
	if len(selected) == 0 {
		return true
	}
	for _, severity := range selected {
		if finding.Severity == severity {
			return true
		}
	}
	return false
}
```

```go
			item.Target,
			item.Summary,
			item.Output,
		}); err != nil {

	if err := w.Error(); err != nil {
		return "", err
	}
	return b.String(), nil
}
```
